424  void TransformedView::DrawSprite(const olc::vf2d & pos, olc::Sprite * sprite, const olc::vf2d & scale, uint8_t flip)
425  {
426  olc::vf2d vSpriteSize = olc::vf2d(float(sprite->width), float(sprite->height));
427  if (IsRectVisible(pos, vSpriteSize * scale))
428  {
429  olc::vf2d vSpriteScaledSize = vSpriteSize * m_vRecipPixel * m_vWorldScale * scale;
430  olc::vi2d vPixel, vStart = WorldToScreen(pos), vEnd = vSpriteScaledSize + vStart;
431  olc::vf2d vPixelStep = 1.0f / vSpriteScaledSize;
432  for (vPixel.y = vStart.y; vPixel.y < vEnd.y; vPixel.y++)
433  {
434  for (vPixel.x = vStart.x; vPixel.x < vEnd.x; vPixel.x++)
435  {
436  olc::vf2d vSample = olc::vf2d(vPixel - vStart) * vPixelStep;
437  pge->Draw(vPixel, sprite->Sample(vSample.x, vSample.y));
438  }
439  }
440  }
441  }
